    es5实现Class和extends

    12/8/2021 语法详解

    # ES6中的Class和extends的底层实现原理

    • Class仅仅只是基于es5现有的原型继承的一种语法糖 要素：
      1. 只能使用new操作符调用Class；
      2. Class可定义实例属性方法和静态属性方法;
    • extends 子类实例可以继承父类上面的属性和方法，子类构造函数可以继承父类的静态属性和静态方法；

    # 源代码

      class Person {
    constructor(options) {
      this.name = options.name
      this.age = options.age
    }
    eat() {
      return 'eating'
    }
    static isPerson(instance) {
      return instance instanceof Person
    }
  }

  class Student extends Person {
    constructor(options) {
      super(options)
      this.grade = options.grade
    }
    study() {
      return 'studying'
    }
    static isStudent(instance) {
      return instance instanceof Student
    }
  }

    # Babel转为es5后的代码(简化)

    var Person = function () {
  function Person(options) {
    // 使用instanceof操作符检测实例是否为指定类的实例（只能使用new操作符调用Class）
    _classCallCheck(this, Person);
    // 实例的属性和方法；
    this.name = options.name;
    this.age = options.age;
  }
    // 给Person的原型上定义法eat && 给Person构造函数定义静态方法isPerson
  _createClass(Person, [{
    key: "eat",
    value: function eat() {
      return 'eating';
    }
  }], [{
    key: "isPerson",
    value: function isPerson(instance) {
      return _instanceof(instance, Person);
    }
  }]);

  return Person;
}();

var Student = function (_Person) {

  _inherits(Student, _Person);
  // 包装了一下父类的构造函数
  var _super = _createSuper(Student);

  function Student(options) {
    var _this;

    _classCallCheck(this, Student);
    // 执行父类构造函数
    _this = _super.call(this, options);
    _this.grade = options.grade;
    return _this;
  }

  _createClass(Student, [{
    key: "study",
    value: function study() {
      return 'studying';
    }
  }], [{
    key: "isStudent",
    value: function isStudent(instance) {
      return _instanceof(instance, Student);
    }
  }]);

  return Student;
}(Person);

        function _inherits(subClass, superClass) {
    //判断superClass（父类构造函数）必须是一个函数或者null
    if (typeof superClass !== "function" && superClass !== null) {
      throw new TypeError("Super expression must either be null or a function");
    }
    // 寄生式继承superClass（父类）的实例属性和方法，更正subClass.prototype的额构造函数为子类；
    // subClass.prototype.__proto__ = superClass.prototype
    subClass.prototype = Object.create(superClass && superClass.prototype, { constructor: { value: subClass, writable: true, configurable: true } });
    // 使用Object.setPrototypeOf()方法继承superClass的静态方法和属性；
    if (superClass) _setPrototypeOf(subClass, superClass);
